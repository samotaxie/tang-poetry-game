    const questions = [

            answer: 2 // 索引從0開始，代表第三個

        },

        {

            ],

        },

        {

            q: "白居易《琵琶行》中，形容樂音「短暫停頓、餘音繞梁」的經典名句是？",

            options: [

                "大弦嘈嘈如急雨，小弦切切如私語。",

                "別有幽愁暗恨生，此時無聲勝有聲。",

            q: "「商女不知亡國恨，隔江猶唱後庭花。」此詩的背景「秦淮河」在哪個城市？",

            options: [

                "長安（西安）",

                "金陵（南京）",

                "洛陽",

                "揚州"

            ],

            answer: 1

        }

    ];


    let currentQuestionIndex = 0;

    let score = 0;


    function startGame() {

        document.getElementById('start-btn').style.display = 'none';

        document.getElementById('quiz-section').style.display = 'block';

        currentQuestionIndex = 0;

        score = 0;

        showQuestion();

    }


    function showQuestion() {

        const progressEl = document.getElementById('progress-text');

        progressEl.innerText = `進度：第 ${currentQuestionIndex + 1} 題 / 共 ${questions.length} 題`;


        const qData = questions[currentQuestionIndex];

        const questionEl = document.getElementById('question-el');

        const optionsEl = document.getElementById('options-el');


        questionEl.innerText = qData.q;

        optionsEl.innerHTML = '';


        qData.options.forEach((opt, index) => {

            const li = document.createElement('li');

            li.className = 'option-item';

            li.innerText = opt;

            li.onclick = () => selectOption(index);

            optionsEl.appendChild(li);

        });

    }


    function selectOption(selectedIndex) {

        const qData = questions[currentQuestionIndex];

        if (selectedIndex === qData.answer) {

            score += 20; // 每題 20 分（共5題）

        }


        currentQuestionIndex++;

        if (currentQuestionIndex < questions.length) {

            showQuestion();

        } else {

            showResult();

        }

    }


    function showResult() {

        document.getElementById('quiz-section').style.display = 'none';

        document.getElementById('result-section').style.display = 'block';

        document.getElementById('progress-text').innerText = '測試完成！';


        const scoreEl = document.getElementById('score-el');

        const suggestionEl = document.getElementById('suggestion-el');


        scoreEl.innerText = `${score} 分`;


        // 給予評分與學習建議

        if (score === 100) {

            suggestionEl.innerText = "太厲害了！你對唐詩三百首的掌握度堪稱「詩仙」級別！對於詩歌的意境、典故與格律都有極佳的敏銳度。建議接下來可以挑戰《全唐詩》中的經典名篇，或是嘗試自己動手創作近體詩，將詩意融入日常生活中。";

        } else if (score >= 80) {

            suggestionEl.innerText = "非常優秀！你已經具備相當紮實的唐詩基礎，能精準辨識經典名句與其背景。美中不足的是少數細節或典故稍有混淆。建議在溫習時，多留意詩人的寫作背景與文學典故（如杜甫的詠懷歷史詩），這能幫你更上一層樓！";

        } else if (score >= 60) {

            suggestionEl.innerText = "及格了！你對著名的唐詩句子有基本的印象，但對於詩句的後半句、詩人情感的轉折或背景理解還不夠穩定。建議可以邊讀邊背，每天挑選一首經典（如王維、孟浩然的山水田園詩），大聲朗讀出聲，培養語感。";

        } else {

            suggestionEl.innerText = "再接再厲！目前對唐詩的熟悉度還有很大的進步空間。別氣餒，詩歌之美需要時間品味。建議先從篇幅較短、畫面感強的五言/七言絕句（如李白、王維的作品）開始，配合翻譯故事來理解，你會發現唐詩其實非常有趣！";

        }

    }


    function resetGame() {

        document.getElementById('result-section').style.display = 'none';

        startGame();

    }
